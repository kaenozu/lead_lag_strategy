### 2. `backtest/real.js`

```javascript
// 連続損失カウンター
let consecutiveLoss = 0;
let positionSize = 1.0;

for (let i = config.warmupPeriod; i < returnsJpOc.length; i++) {
  // ... シグナル計算 ...
  
  // 連続損失カウント
  if (strategyRet < 0) {
    consecutiveLoss++;
    if (consecutiveLoss >= config.consecutiveLoss.threshold) {
      positionSize = 1.0 - config.consecutiveLoss.reduction;
    }
  } else {
    consecutiveLoss = 0;
    positionSize = 1.0;
  }
  
  // ポジションサイズ適用
  strategyRet *= positionSize;
}
```
